Build the Backlog — Console Debugging Guide

Copy-paste snippets for the browser DevTools console. They read the live in-memory games array — the same data currently rendered on screen — so open the app, let it finish loading, then open DevTools (F12 or Cmd/Ctrl+Shift+J), go to the Console tab, paste a snippet, and hit Enter. Every snippet is self-contained: no setup step, no shared state between them.

These are read-only reporting scripts — none of them modify games, write to the sheet, or touch localStorage. Safe to run at any time, including on production data.

Unique Genres

Every distinct value in games[].genres, with how many games use each, sorted alphabetically.

Unique genres + counts
(() => {
  const toArr = v => Array.isArray(v) ? v : (typeof v === 'string' && v.trim() ? v.split(',').map(s => s.trim()).filter(Boolean) : []);
  const counts = {};
  games.forEach(g => toArr(g.genres).forEach(v => { if (v) counts[v] = (counts[v] || 0) + 1; }));
  const rows = Object.entries(counts).sort((a, b) => a[0].localeCompare(b[0])).map(([genre, count]) => ({ genre, count }));
  console.table(rows);
  console.log(`${rows.length} unique genres · ${games.length} games scanned`);
})();

Casing / Spacing Variants taxonomy hygiene

Flags values that are duplicates in spirit but not in exact string form — e.g. "Co-op" vs "co-op", or a stray trailing space. These fragment filters silently, so they're worth cleaning up even though they aren't literal duplicates.

Find case/whitespace variant groups
(() => {
  const toArr = v => Array.isArray(v) ? v : (typeof v === 'string' && v.trim() ? v.split(',').map(s => s.trim()).filter(Boolean) : []);
  let found = false;
  ['genres', 'tags', 'developer', 'publisher'].forEach(field => {
    const variants = {};
    games.forEach(g => toArr(g[field]).forEach(v => {
      if (!v) return;
      const key = v.toLowerCase();
      (variants[key] = variants[key] || new Set()).add(v);
    }));
    const clashes = Object.entries(variants).filter(([, set]) => set.size > 1);
    if (clashes.length) {
      found = true;
      console.group(`⚠ ${field} — ${clashes.length} variant group(s)`);
      clashes.forEach(([, set]) => console.log([...set].join('  ⟷  ')));
      console.groupEnd();
    }
  });
  if (!found) console.log('No casing/spacing variants found across genres, tags, developer, publisher.');
})();

Duplicate Games — by Title

Groups games whose titles match case-insensitively (trimmed), and prints each group with its id, status, steamAppId and type so you can tell which entry to keep.

Find duplicate titles
(() => {
  const groups = {};
  games.forEach(g => {
    const key = (g.title || '').trim().toLowerCase();
    if (!key) return;
    (groups[key] = groups[key] || []).push(g);
  });
  const dupes = Object.values(groups).filter(arr => arr.length > 1);
  console.log(`${dupes.length} duplicate title group(s)`);
  dupes.forEach(arr => {
    console.group(`"${arr[0].title}" × ${arr.length}`);
    console.table(arr.map(g => ({ id: g.id, status: g.status, steamAppId: g.steamAppId, type: g.type })));
    console.groupEnd();
  });
})();

Duplicate Games — by Steam App ID

Groups games sharing the same non-empty steamAppId. Two entries with the same App ID are almost always the same game added twice (e.g. once on the wishlist, once accidentally re-added after purchase).

Find duplicate Steam App IDs
(() => {
  const groups = {};
  games.forEach(g => {
    const id = g.steamAppId ? String(g.steamAppId).trim() : '';
    if (!id) return;
    (groups[id] = groups[id] || []).push(g);
  });
  const dupes = Object.entries(groups).filter(([, arr]) => arr.length > 1);
  console.log(`${dupes.length} duplicate steamAppId group(s)`);
  dupes.forEach(([id, arr]) => {
    console.group(`steamAppId ${id} × ${arr.length}`);
    console.table(arr.map(g => ({ id: g.id, title: g.title, status: g.status, type: g.type })));
    console.groupEnd();
  });
})();

Orphan DLC integrity check

DLC entries (type: 'dlc') whose parentAppId doesn't match the steamAppId of any game in the list — these won't nest under a parent and will render as if they had no parent.

Find DLC with no matching parent
(() => {
  const appIds = new Set(games.filter(g => g.steamAppId).map(g => String(g.steamAppId)));
  const orphans = games.filter(g => g.type === 'dlc' && g.parentAppId && !appIds.has(String(g.parentAppId)));
  console.log(`${orphans.length} orphan DLC entr${orphans.length === 1 ? 'y' : 'ies'}`);
  console.table(orphans.map(g => ({ id: g.id, title: g.title, parentAppId: g.parentAppId, status: g.status })));
})();

Missing Data

Non-cancelled games missing a Steam App ID, cover image, genres, developer, or publisher — the fields that drive cover art, filters, and store links. DLC rows are excluded from the Steam App ID check since some DLC is tracked without its own store page.

Find incomplete game records
(() => {
  const toArr = v => Array.isArray(v) ? v : (typeof v === 'string' && v.trim() ? v.split(',').map(s => s.trim()).filter(Boolean) : []);
  const rows = games.filter(g => g.status !== 'cancelled').map(g => {
    const missing = [];
    if (!g.steamAppId && g.type !== 'dlc') missing.push('steamAppId');
    if (!g.cover && !g.steamAppId) missing.push('cover');
    if (!toArr(g.genres).length) missing.push('genres');
    if (!toArr(g.developer).length) missing.push('developer');
    if (!toArr(g.publisher).length) missing.push('publisher');
    return missing.length ? { id: g.id, title: g.title, status: g.status, missing: missing.join(', ') } : null;
  }).filter(Boolean);
  console.log(`${rows.length} game(s) with missing data`);
  console.table(rows);
})();

Trade Keys

Markdown list of every game with at least one spare trading key set (games[].key, a JSON array of strings populated by hand in the synced Google Sheet), as [title](steamURL) — paste straight into a Markdown doc or Discord message.

Markdown list of games with a trade key
(() => {
  const steamUrl = g => g.storeLink || (g.steamAppId
    ? `https://store.steampowered.com/app/${g.steamAppId}/`
    : `https://store.steampowered.com/search/?term=${encodeURIComponent(g.title || '')}`);
  const withKey = games.filter(g => g.key && g.key.length);
  const md = withKey.map(g => `[${g.title}](${steamUrl(g)})`).join('\n');
  console.log(md || 'No games with a trade key set.');
  console.log(`${withKey.length} game(s) with a trade key`);
})();

Full Report

Runs every check above in one paste, prints a grouped summary to the console, and also stashes the raw result on window._btbReport so you can inspect it further (e.g. _btbReport.dupeTitles).

Run everything
(() => {
  const toArr = v => Array.isArray(v) ? v : (typeof v === 'string' && v.trim() ? v.split(',').map(s => s.trim()).filter(Boolean) : []);

  const taxonomy = {};
  const variants = {};
  ['genres', 'tags', 'developer', 'publisher'].forEach(field => {
    const counts = {};
    const seen = {};
    games.forEach(g => toArr(g[field]).forEach(v => {
      if (!v) return;
      counts[v] = (counts[v] || 0) + 1;
      const key = v.toLowerCase();
      (seen[key] = seen[key] || new Set()).add(v);
    }));
    taxonomy[field] = Object.entries(counts).sort((a, b) => a[0].localeCompare(b[0]));
    variants[field] = Object.values(seen).filter(set => set.size > 1).map(set => [...set]);
  });

  const titleGroups = {};
  games.forEach(g => {
    const key = (g.title || '').trim().toLowerCase();
    if (key) (titleGroups[key] = titleGroups[key] || []).push(g);
  });
  const dupeTitles = Object.values(titleGroups).filter(arr => arr.length > 1);

  const appIdGroups = {};
  games.forEach(g => {
    const id = g.steamAppId ? String(g.steamAppId).trim() : '';
    if (id) (appIdGroups[id] = appIdGroups[id] || []).push(g);
  });
  const dupeAppIds = Object.entries(appIdGroups).filter(([, arr]) => arr.length > 1);

  const appIds = new Set(games.filter(g => g.steamAppId).map(g => String(g.steamAppId)));
  const orphanDlc = games.filter(g => g.type === 'dlc' && g.parentAppId && !appIds.has(String(g.parentAppId)));

  const missingData = games.filter(g => g.status !== 'cancelled').map(g => {
    const missing = [];
    if (!g.steamAppId && g.type !== 'dlc') missing.push('steamAppId');
    if (!g.cover && !g.steamAppId) missing.push('cover');
    if (!toArr(g.genres).length) missing.push('genres');
    if (!toArr(g.developer).length) missing.push('developer');
    if (!toArr(g.publisher).length) missing.push('publisher');
    return missing.length ? { id: g.id, title: g.title, status: g.status, missing } : null;
  }).filter(Boolean);

  window._btbReport = { taxonomy, variants, dupeTitles, dupeAppIds, orphanDlc, missingData };

  console.group(`%cBuild the Backlog — Full Report (${games.length} games)`, 'color:#02b3fc;font-weight:bold');
  ['genres', 'tags', 'developer', 'publisher'].forEach(field => {
    console.group(`${field} — ${taxonomy[field].length} unique`);
    console.table(taxonomy[field].map(([name, count]) => ({ [field]: name, count })));
    if (variants[field].length) console.log(`⚠ ${variants[field].length} casing/spacing variant group(s):`, variants[field]);
    console.groupEnd();
  });
  console.group(`Duplicate titles — ${dupeTitles.length}`);
  dupeTitles.forEach(arr => console.log(arr[0].title, '×', arr.length, arr.map(g => g.id)));
  console.groupEnd();
  console.group(`Duplicate Steam App IDs — ${dupeAppIds.length}`);
  dupeAppIds.forEach(([id, arr]) => console.log(id, '×', arr.length, arr.map(g => g.title)));
  console.groupEnd();
  console.group(`Orphan DLC — ${orphanDlc.length}`);
  console.table(orphanDlc.map(g => ({ id: g.id, title: g.title, parentAppId: g.parentAppId })));
  console.groupEnd();
  console.group(`Missing data — ${missingData.length}`);
  console.table(missingData.map(r => ({ ...r, missing: r.missing.join(', ') })));
  console.groupEnd();
  console.groupEnd();
  console.log('Full result stored on window._btbReport');
})();
